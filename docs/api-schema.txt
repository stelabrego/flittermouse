POST /user/add/ creates new user
    req
    {
        username: string
        email: string
        password: string
        inviteKey: string TODO
    }
    res
    {
        status: int
        message: string
        userKey: string optional
    }

PUT /users/update updates user
    req {
        userKey: string
        username: string optional
        email: string optional
        password: string optional
        displayName: string optional
        phoneNumber: string optional
        address: string optional
        bio: string optional
    }
    res
    {
        status: int
        message: string
    }

DELETE /users/delete deletes user
    req {
        userKey: string
    }
    res
    {
        status: int
        message: string
    }

POST /event/add/ creates new event
    req
    {
        name: string
        userKey: string
        location: string optional
        dateOf: string optional
    }
    res
    {
        status: int
        message: string
        eventKey: string optional
    }

DELETE /events/delete deletes event
    req {
        eventKey: string
    }
    res
    {
        status: int
        message: string
    }



GET / returns homepage
GET /:username returns user profile page
GET /users/:userKey redirects to /:username
GET /signup returns "sorry invite only rn"
GET /signup/:inviteKey returns signup page, let user know if inviteKey is valid
GET /events/:eventKey returns event page
